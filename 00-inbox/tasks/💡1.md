```dataviewjs
const startOfWeek = moment().startOf('week');
const endOfWeek = moment().endOf('week');

const weekPages = dv.pages('"Daily Notes"')
    .where(p => moment(p.date).isBetween(startOfWeek, endOfWeek, null, '[]'))
    .sort(p => p.date, 'desc');

dv.header(3, "Training Summary");
dv.table(
    ["Date", "Workout", "Load (RPE)", "Energy", "Recovery"],
    weekPages.map(p => [
        p.date,
        p.workout_type || "Rest",
        p.training_load || "-",
        p.energy || "-",
        p.recovery_score || "-"
    ])
);

// Calculate weekly volume
const totalSessions = weekPages.filter(p => p.workout_completed).length;
const avgLoad = weekPages.filter(p => p.training_load).array()
    .reduce((sum, p) => sum + p.training_load, 0) / totalSessions || 0;

dv.header(4, `Sessions: ${totalSessions} | Avg Load: ${avgLoad.toFixed(1)}/10`);
```

## 🎯 Monthly Progress

```dataviewjs
const last30Days = dv.pages('"Daily Notes"')
    .where(p => moment(p.date).isAfter(moment().subtract(30, 'days')))
    .filter(p => p.workout_completed);

const workoutCount = last30Days.length;
const avgEnergy = last30Days.array()
    .reduce((sum, p) => sum + (p.energy || 0), 0) / workoutCount || 0;

dv.paragraph(`**30-Day Stats:**
- Total Workouts: ${workoutCount}
- Avg Energy: ${avgEnergy.toFixed(1)}/10
- Consistency: ${((workoutCount / 30) * 100).toFixed(0)}%`);
```

## 🔥 Current Streak

```dataviewjs
let streak = 0;
const pages = dv.pages('"Daily Notes"')
    .sort(p => p.date, 'desc')
    .array();

for (let p of pages) {
    if (p.workout_completed) streak++;
    else break;
}

dv.header(3, `🔥 ${streak} Day Workout Streak`);
```
